\NewSseqCommand\twoclass{O{}d()}{
    \IfNoValueF{#2}{\pushstack(#2)}
    \class[#1](\lastx,\lasty+1)
    \structline
}

\NewSseqCommand\divtwoclass{O{}d()}{
    \IfNoValueF{#2}{\pushstack(#2)}
    \class[#1](\lastx,\lasty-1)
    \structline
}
